30  public:
31  ExternalEditor()
32  : externalEditorPid(0),
33  launchX(0),
34  launchY(0),
35  launchWidth(0),
36  launchHeight(0),
37  dirty(true),
38  hasReceivedFirstMessage(false) {
39  editorBridge.OnReceive(
40  [this](gd::String cmd, gd::SerializerElement object, gd::String scope) {
41  if (!hasReceivedFirstMessage) {
42  hasReceivedFirstMessage = true;
43  if (onLaunchedCb) onLaunchedCb();
44  }
45 
46  if (cmd == "update") {
47  if (onUpdateReceivedCb) onUpdateReceivedCb(object, scope);
48  } else if (cmd == "requestUpdate")
49  SendUpdate(scope);
50  else if (cmd == "requestForcedUpdate")
51  SendUpdate(scope, true);
52  else if (cmd == "requestPreview") {
53  if (onLaunchPreviewCb) onLaunchPreviewCb();
54  } else if (cmd == "editObject") {
55  if (onEditObjectCb) onEditObjectCb(object.GetValue().GetString());
56  } else
57  std::cout << "Received message with unknown command: \"" << cmd
58  << "\"" << std::endl;
59  });
60  }
