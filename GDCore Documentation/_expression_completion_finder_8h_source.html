175 class GD_CORE_API ExpressionCompletionFinder
176  : public ExpressionParser2NodeWorker {
177  public:
182  static std::vector<ExpressionCompletionDescription>
183  GetCompletionDescriptionsFor(gd::ExpressionNode& node,
184  size_t searchedPosition) {
185  gd::ExpressionNode* nodeAtLocation =
186  gd::ExpressionNodeLocationFinder::GetNodeAtPosition(node,
187  searchedPosition);
188 
189  if (nodeAtLocation == nullptr) {
190  std::vector<ExpressionCompletionDescription> emptyCompletions;
191  return emptyCompletions;
192  }
193 
194  gd::ExpressionCompletionFinder autocompletionProvider(searchedPosition);
195  nodeAtLocation->Visit(autocompletionProvider);
196  return autocompletionProvider.GetCompletionDescriptions();
197  }
198 
202  const std::vector<ExpressionCompletionDescription>&
203  GetCompletionDescriptions() {
204  return completions;
205  };
206 
207  virtual ~ExpressionCompletionFinder(){};
208 
209  protected:
210  void OnVisitSubExpressionNode(SubExpressionNode& node) override {
211  completions.push_back(
212  ExpressionCompletionDescription::ForObject(node.type, ""));
213  completions.push_back(
214  ExpressionCompletionDescription::ForExpression(node.type, ""));
215  }
216  void OnVisitOperatorNode(OperatorNode& node) override {
217  completions.push_back(
218  ExpressionCompletionDescription::ForObject(node.type, ""));
219  completions.push_back(
220  ExpressionCompletionDescription::ForExpression(node.type, ""));
221  }
222  void OnVisitUnaryOperatorNode(UnaryOperatorNode& node) override {
223  completions.push_back(
224  ExpressionCompletionDescription::ForObject(node.type, ""));
225  completions.push_back(
226  ExpressionCompletionDescription::ForExpression(node.type, ""));
227  }
228  void OnVisitNumberNode(NumberNode& node) override {
229  // No completions
230  }
231  void OnVisitTextNode(TextNode& node) override {
232  // No completions
233  }
234  void OnVisitVariableNode(VariableNode& node) override {
235  completions.push_back(
236  ExpressionCompletionDescription::ForVariable(node.type, node.name));
237  }
238  void OnVisitVariableAccessorNode(VariableAccessorNode& node) override {
239  // No completions
240  }
241  void OnVisitVariableBracketAccessorNode(
242  VariableBracketAccessorNode& node) override {
243  // No completions
244  }
245  void OnVisitIdentifierNode(IdentifierNode& node) override {
246  if (gd::ParameterMetadata::IsObject(node.type)) {
247  // Only show completions of objects if an object is required
248  completions.push_back(ExpressionCompletionDescription::ForObject(
249  node.type, node.identifierName));
250  } else {
251  // Show completions for expressions and objects otherwise.
252  completions.push_back(ExpressionCompletionDescription::ForObject(
253  node.type, node.identifierName));
254  completions.push_back(ExpressionCompletionDescription::ForExpression(
255  node.type, node.identifierName));
256  }
257  }
258  void OnVisitObjectFunctionNameNode(ObjectFunctionNameNode& node) override {
259  if (!node.behaviorFunctionName.empty() ||
260  node.behaviorNameNamespaceSeparatorLocation.IsValid()) {
261  // Behavior function (or behavior function being written, with the
262  // function name missing)
263  if (IsCaretOn(node.objectNameLocation)) {
264  completions.push_back(ExpressionCompletionDescription::ForObject(
265  node.type, node.objectName));
266  } else if (IsCaretOn(node.objectNameDotLocation) ||
267  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
268  completions.push_back(ExpressionCompletionDescription::ForBehavior(
269  node.objectFunctionOrBehaviorName, node.objectName));
270  } else if (IsCaretOn(node.behaviorNameNamespaceSeparatorLocation) ||
271  IsCaretOn(node.behaviorFunctionNameLocation)) {
272  completions.push_back(ExpressionCompletionDescription::ForExpression(
273  node.type,
274  node.behaviorFunctionName,
275  node.objectName,
276  node.objectFunctionOrBehaviorName));
277  }
278  } else {
279  // Object function or behavior name
280  if (IsCaretOn(node.objectNameLocation)) {
281  completions.push_back(ExpressionCompletionDescription::ForObject(
282  node.type, node.objectName));
283  } else if (IsCaretOn(node.objectNameDotLocation) ||
284  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
285  completions.push_back(ExpressionCompletionDescription::ForBehavior(
286  node.objectFunctionOrBehaviorName, node.objectName));
287  completions.push_back(ExpressionCompletionDescription::ForExpression(
288  node.type, node.objectFunctionOrBehaviorName, node.objectName));
289  }
290  }
291  }
292  void OnVisitFunctionCallNode(FunctionCallNode& node) override {
293  bool isCaretOnParenthesis = IsCaretOn(node.openingParenthesisLocation) ||
294  IsCaretOn(node.closingParenthesisLocation);
295 
296  if (!node.behaviorName.empty()) {
297  // Behavior function
298  if (IsCaretOn(node.objectNameLocation)) {
299  completions.push_back(ExpressionCompletionDescription::ForObject(
300  node.type, node.objectName));
301  } else if (IsCaretOn(node.objectNameDotLocation) ||
302  IsCaretOn(node.behaviorNameLocation)) {
303  completions.push_back(ExpressionCompletionDescription::ForBehavior(
304  node.behaviorName, node.objectName));
305  } else {
306  completions.push_back(
307  ExpressionCompletionDescription::ForExpression(node.type,
308  node.functionName,
309  node.objectName,
310  node.behaviorName)
311  .SetIsExact(isCaretOnParenthesis));
312  }
313  } else if (!node.objectName.empty()) {
314  // Object function
315  if (IsCaretOn(node.objectNameLocation)) {
316  completions.push_back(ExpressionCompletionDescription::ForObject(
317  node.type, node.objectName));
318  } else {
319  // Add completions for behaviors, because we could imagine that the user
320  // wants to move from an object function to a behavior function, and so
321  // need behavior completions. Do this unless we're on the parenthesis
322  // (at which point we're only showing informative message about the
323  // function).
324  if (!isCaretOnParenthesis) {
325  completions.push_back(ExpressionCompletionDescription::ForBehavior(
326  node.functionName, node.objectName));
327  }
328 
329  completions.push_back(ExpressionCompletionDescription::ForExpression(
330  node.type, node.functionName, node.objectName)
331  .SetIsExact(isCaretOnParenthesis));
332  }
333  } else {
334  // Free function
335  completions.push_back(ExpressionCompletionDescription::ForExpression(
336  node.type, node.functionName)
337  .SetIsExact(isCaretOnParenthesis));
338  }
339  }
340  void OnVisitEmptyNode(EmptyNode& node) override {
341  completions.push_back(
342  ExpressionCompletionDescription::ForObject(node.type, node.text));
343  completions.push_back(
344  ExpressionCompletionDescription::ForExpression(node.type, node.text));
345  }
346 
347  private:
348  bool IsCaretOn(const ExpressionParserLocation& location,
349  bool inclusive = false) {
350  if (!location.IsValid()) return false;
351 
352  return (location.GetStartPosition() <= searchedPosition &&
353  ((!inclusive && searchedPosition < location.GetEndPosition()) ||
354  (inclusive && searchedPosition <= location.GetEndPosition())));
355  }
356 
357  ExpressionCompletionFinder(size_t searchedPosition_)
358  : searchedPosition(searchedPosition_){};
359 
360  std::vector<ExpressionCompletionDescription> completions;
361  size_t searchedPosition;
362 };
