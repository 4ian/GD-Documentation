273 struct hash<gd::EventItem> {
274  std::size_t operator()(gd::EventItem const& item) const {
275  return (std::hash<gd::BaseEvent*>()(item.event.get())) ^
276  (std::hash<gd::EventsList*>()(item.eventsList) << 1) ^
277  (std::hash<std::size_t>()(item.positionInList) << 2);
278  }
279 };

281 template <>
282 struct hash<gd::InstructionItem> {
283  std::size_t operator()(gd::InstructionItem const& item) const {
284  return (std::hash<gd::Instruction*>()(item.instruction)) ^
285  (std::hash<gd::InstructionsList*>()(item.instructionList) << 1) ^
286  (std::hash<std::size_t>()(item.positionInList) << 2) ^
287  (std::hash<gd::BaseEvent*>()(item.event) << 3) ^
288  (std::hash<bool>()(item.isCondition) << 4);
289  }
290 };

292 template <>

294  std::size_t operator()(gd::InstructionListItem const& item) const {
295  return (std::hash<gd::InstructionsList*>()(item.instructionList)) ^
296  (std::hash<gd::BaseEvent*>()(item.event) << 1) ^
297  (std::hash<bool>()(item.isConditionList) << 2);
