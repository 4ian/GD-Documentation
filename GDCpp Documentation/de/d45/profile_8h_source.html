108  unsigned long int getTimeMilliseconds()
109  {
110 #ifdef USE_WINDOWS_TIMERS
111  LARGE_INTEGER currentTime;
112  QueryPerformanceCounter(&currentTime);
113  LONGLONG elapsedTime = currentTime.QuadPart -
114  mStartTime.QuadPart;
115 
116  // Compute the number of millisecond ticks elapsed.
117  unsigned long msecTicks = (unsigned long)(1000 * elapsedTime /
118  mClockFrequency.QuadPart);
119 
120  // Check for unexpected leaps in the Win32 performance counter.
121  // (This is caused by unexpected data across the PCI to ISA
122  // bridge, aka south bridge. See Microsoft KB274323.)
123  unsigned long elapsedTicks = GetTickCount() - mStartTick;
124  signed long msecOff = (signed long)(msecTicks - elapsedTicks);
125  if (msecOff < -100 || msecOff > 100)
126  {
127  // Adjust the starting time forwards.
128  LONGLONG msecAdjustment = mymin(msecOff *
129  mClockFrequency.QuadPart / 1000, elapsedTime -
130  mPrevElapsedTime);
131  mStartTime.QuadPart += msecAdjustment;
132  elapsedTime -= msecAdjustment;
133 
134  // Recompute the number of millisecond ticks elapsed.
135  msecTicks = (unsigned long)(1000 * elapsedTime /
136  mClockFrequency.QuadPart);
137  }
138 
139  // Store the current elapsed time for adjustments next time.
140  mPrevElapsedTime = elapsedTime;
141 
142  return msecTicks;
143 #else
144 
145 #ifdef __CELLOS_LV2__
146  uint64_t freq=sys_time_get_timebase_frequency();
147  double dFreq=((double) freq) / 1000.0;
148  typedef uint64_t ClockSize;
149  ClockSize newTime;
150  SYS_TIMEBASE_GET( newTime );
151  //__asm __volatile__( "mftb %0" : "=r" (newTime) : : "memory");
152 
153  return (unsigned long int)((double(newTime-mStartTime)) / dFreq);
154 #else
155 
156  struct timeval currentTime;
157  gettimeofday(&currentTime, 0);
158  return (currentTime.tv_sec - mStartTime.tv_sec) * 1000 +
159  (currentTime.tv_usec - mStartTime.tv_usec) / 1000;
160 #endif //__CELLOS_LV2__
161 #endif
162  }
